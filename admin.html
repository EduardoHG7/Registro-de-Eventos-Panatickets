⚙ Configuración Inicial — Conectar Google Sheets
  1. Abre sheets.new y crea una hoja llamada Registros
  2. En el menú ve a Extensiones → Apps Script
  3. Borra el contenido y pega el siguiente script:
function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('Registros'); var data = JSON.parse(e.postData.contents); var now = data.fechaHora; // Fila del socio sheet.appendRow([ now, data.nombre, data.apellido, data.numSocio, data.numMesa, data.tieneInvitados, data.cantInvitados, '', '', '' ]); // Filas de invitados if (data.invitados && data.invitados.length > 0) { data.invitados.forEach(function(inv) { sheet.appendRow([ now, data.nombre + ' ' + data.apellido, '(invitado)', data.numSocio, data.numMesa, '', '', inv.nombre, inv.apellido, inv.genero ]); }); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('Registros'); var rows = sheet.getDataRange().getValues(); return ContentService .createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); }
